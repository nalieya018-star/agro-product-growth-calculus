    <script>
        // Initialization & DOM Selectors
        let chart;
        let animationInterval;
        
        // Element Hooks
        const paramL = document.getElementById('paramL');
        const paramK = document.getElementById('paramK');
        const paramX0 = document.getElementById('paramX0');

        const valL = document.getElementById('valL');
        const valK = document.getElementById('valK');
        const valX0 = document.getElementById('valX0');

        const outFx = document.getElementById('outFx');
        const outFpx = document.getElementById('outFpx');
        const laymanText = document.getElementById('laymanText');
        const stageBadge = document.getElementById('stageBadge');
        const fruitletPath = document.getElementById('fruitletPath');
        const fruitletDAA = document.getElementById('fruitletDAA');
        
        // Mathematical Logic Functions
        function getModelParams() {
            return {
                L: parseFloat(paramL.value),
                k: parseFloat(paramK.value),
                x0: parseFloat(paramX0.value)
            };
        }

        // Logistic Cumulative Function: f(x)
        function f(x, L, k, x0) {
            return L / (1 + Math.exp(-k * (x - x0)));
        }

        // Analytical First Derivative Function: f'(x)
        function fPrime(x, L, k, x0) {
            const expTerm = Math.exp(-k * (x - x0));
            const denominator = Math.pow(1 + expTerm, 2);
            return (L * k * expTerm) / denominator;
        }

        // Generate entire range of data from x = 100 to 180 DAA
        function generateDataRange() {
            const labels = [];
            const fData = [];
            const fPrimeData = [];
            const { L, k, x0 } = getModelParams();

            for (let x = 100; x <= 180; x++) {
                labels.push(x);
                fData.push(f(x, L, k, x0).toFixed(2));
                fPrimeData.push(fPrime(x, L, k, x0).toFixed(3));
            }
            return { labels, fData, fPrimeData };
        }

        // Update Slider Numerical Outputs
        function updateSliderLabels() {
            valL.textContent = `${paramL.value}%`;
            valK.textContent = paramK.value;
            valX0.textContent = `${paramX0.value} DAA`;
        }

        // Dynamically update the visual rendering of the fruitlet SVG
        function updateFruitletVisual(daa, pct) {
            fruitletDAA.textContent = daa;
            
            // Color Interpolation for Malaysian Palm Oil Fruit (Deep Violet-Black -> Orange-Red -> Orange-Yellow)
            let fillColor;
            if (daa < 125) {
                // Unripe / Lag Stage (Deep Black/Purple)
                fillColor = '#1e0f35';
            } else if (daa >= 125 && daa < 145) {
                // Turning / Accelerated Stage (Reddish Black/Brown)
                fillColor = '#5c171e';
            } else if (daa >= 145 && daa < 160) {
                // Ripe / High Biosynthesis Stage (Fiery Red-Orange)
                fillColor = '#d33611';
            } else {
                // Fully Mature / Plateau Stage (Deep Vibrant Orange-Yellow base)
                fillColor = '#e65c00';
            }
            
            fruitletPath.setAttribute('fill', fillColor);
        }

        // Stage Assessment & Layman Agronomic Explanations
        function getStageAnalysis(x, fx, fpx) {
            const { L, k, x0 } = getModelParams();
            let badgeClass = '';
            let stageName = '';
            let description = '';

            if (x < 120) {
                stageName = "Unripe / Lag Phase";
                badgeClass = "bg-purple-100 text-purple-800 border-purple-200";
                description = `At ${x} Days After Anthesis (DAA), the mesocarp is in early development. Lipid synthesis rate is extremely low (${fpx}% per day) because energy is channeled into core cellular structure development. Total oil content is tiny (${fx}%). Processing this bunch yields almost zero oil. DO NOT HARVEST.`;
            } 
            else if (x >= 120 && x < (x0 - 5)) {
                stageName = "Acceleration Phase";
                badgeClass = "bg-blue-100 text-blue-800 border-blue-200";
                description = `Between 120 DAA and ${Math.round(x0 - 5)} DAA, enzymes responsible for fatty acid synthesis (e.g., SAD, FATA) are heavily activated. Cumulative oil is at ${fx}%, accumulating aggressively at ${fpx}% daily. Harvesting now results in sub-optimal oil extraction and lower commercial grade.`;
            } 
            else if (x >= (x0 - 5) && x <= (x0 + 5)) {
                stageName = "Peak Synthesis Window";
                badgeClass = "bg-amber-100 text-amber-800 border-amber-200";
                description = `Peak biosynthesis rate achieved! Currently accumulating lipid at its near-maximum rate of ${fpx}% per day. The mesocarp cells are actively packaging lipids into oil droplets. Harvesting at exactly ${x} DAA is not recommended as substantial synthesis capacity remains.`;
            } 
            else if (x > (x0 + 5) && x < 160) {
                stageName = "Deceleration Phase";
                badgeClass = "bg-emerald-100 text-emerald-800 border-emerald-200";
                description = `Oil synthesis rate starts tapering down to ${fpx}% per day as mesocarp cells run out of storage space. However, cumulative oil has built up nicely to ${fx}%. Harvesting is approaching, but waiting a bit longer will yield more extractable CPO.`;
            } 
            else {
                stageName = "Commercial Plateau / Ideal Harvest";
                badgeClass = "bg-rose-100 text-rose-800 border-rose-200";
                description = `Optimal harvesting window! At ${x} DAA, the rate of synthesis has bottomed out to ${fpx}% per day because mesocarp cells are saturated. Cumulative oil has plateaued at ${fx}%. Harvesting must proceed immediately to prevent over-ripening and spikes in Free Fatty Acid (FFA) values.`;
            }

            return { stageName, badgeClass, description };
        }

        // Handle Dynamic Interactivity on Specific Data Points
        function inspectPoint(daa) {
            const { L, k, x0 } = getModelParams();
            const calculatedFx = f(daa, L, k, x0).toFixed(2);
            const calculatedFpx = fPrime(daa, L, k, x0).toFixed(3);

            // Populate Output Metrics
            outFx.textContent = `${calculatedFx}%`;
            outFpx.textContent = `${calculatedFpx}%/day`;

            // Dynamic Styling Updates
            const analysis = getStageAnalysis(daa, calculatedFx, calculatedFpx);
            stageBadge.className = `inline-block text-xs font-bold px-3 py-1 rounded-full border ${analysis.badgeClass}`;
            stageBadge.textContent = analysis.stageName;
            laymanText.innerHTML = analysis.description;

            // Visual Update to SVG Fruit
            updateFruitletVisual(daa, parseFloat(calculatedFx));

            // Custom Chart Reticle highlight
            if (chart) {
                const chartPoints = chart.data.labels;
                const pointIndex = chartPoints.indexOf(daa);
                if (pointIndex !== -1) {
                    chart.setActiveElements([{
                        datasetIndex: 0,
                        index: pointIndex
                    }]);
                    chart.update();
                }
            }
        }

        // Initialize and Configure Interactive Chart.js
        function buildChart() {
            const ctx = document.getElementById('growthChart').getContext('2d');
            const dataSets = generateDataRange();

            chart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: dataSets.labels,
                    datasets: [
                        {
                            label: 'Cumulative Oil Content f(x) (%)',
                            data: dataSets.fData,

            let currentDaa = 100;
            inspectPoint(currentDaa);

            animationInterval = setInterval(() => {
                currentDaa++;
                inspectPoint(currentDaa);

                if (currentDaa >= 180) {
                    clearInterval(animationInterval);
                }
            }, 80); // Quick incremental step animation
